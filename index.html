        function showApp() {

        // 即時監聽資料庫變化

        function listenToData() {

            onSnapshot(tenantsCollection, (snapshot) => {

                const listData = [];

                snapshot.forEach((doc) => {

                    listData.push({ id: doc.id, ...doc.data() });

                });

                renderTable(listData);

            });

        }


        // 渲染表格與自動判斷「遲繳」

        function renderTable(data) {

            const tbody = document.getElementById('tenant-list');

            tbody.innerHTML = '';

            

            if(data.length === 0) {

                tbody.innerHTML = `<tr><td colspan="7" class="p-8 text-sm text-gray-400 text-center">目前沒有記錄，請從上方新增。</td></tr>`;

                return;

            }


            const todayStr = new Date().toISOString().split('T')[0]; // 取得今天日期字串 (YYYY-MM-DD)


            data.forEach(item => {

                let statusBadge = '';

                

                // 核心邏輯：如果狀態是未繳，且今天大於繳費日期，自動判定為遲繳

                if (item.status === '已繳') {

                    statusBadge = `<span class="inline-flex items-center gap-1 px-2.5 py-1 text-xs font-medium bg-emerald-50 text-emerald-700 rounded-md"><span class="w-1.5 h-1.5 rounded-full bg-emerald-500"></span>已繳</span>`;

                } else if (item.status === '未繳' && todayStr > item.dueDate) {

                    statusBadge = `<span class="inline-flex items-center gap-1 px-2.5 py-1 text-xs font-medium bg-rose-50 text-rose-700 rounded-md animate-pulse"><span class="w-1.5 h-1.5 rounded-full bg-rose-500"></span>遲繳 ⚠️</span>`;

                } else {

                    statusBadge = `<span class="inline-flex items-center gap-1 px-2.5 py-1 text-xs font-medium bg-amber-50 text-amber-700 rounded-md"><span class="w-1.5 h-1.5 rounded-full bg-amber-500"></span>未繳</span>`;

                }


                tbody.innerHTML += `

                    <tr class="hover:bg-gray-50/50 transition">

                        <td class="p-4 text-sm font-semibold text-gray-900">${item.name}</td>

                        <td class="p-4 text-sm text-gray-600">$${Number(item.weeklyRent).toLocaleString()}</td>

                        <td class="p-4 text-sm text-gray-500">${item.moveInDate || '-'}</td>

                        <td class="p-4 text-sm font-medium text-gray-700">${item.dueDate}</td>

                        <td class="p-4 text-sm">${statusBadge}</td>

                        <td class="p-4 text-sm text-gray-400 max-w-xs truncate" title="${item.remark || ''}">${item.remark || '-'}</td>

                        <td class="p-4 text-sm text-right space-x-2">

                            <button onclick="editTenant('${item.id}', '${item.name}', '${item.weeklyRent}', '${item.moveInDate}', '${item.dueDate}', '${item.status}', '${item.remark || ''}')" class="text-gray-400 hover:text-gray-900 transition"><i class="fa-regular fa-pen-to-square"></i></button>

                            <button onclick="deleteTenant('${item.id}')" class="text-gray-400 hover:text-rose-600 transition"><i class="fa-regular fa-trash-can"></i></button>

                        </td>

                    </tr>

                `;

            });

        }


        // 新增或更新資料

        async function saveTenant() {

            const id = document.getElementById('edit-id').value;

            const name = document.getElementById('tenant-name').value.trim();

            const weeklyRent = document.getElementById('weekly-rent').value;

            const moveInDate = document.getElementById('move-in-date').value;

            const dueDate = document.getElementById('due-date').value;

            const status = document.getElementById('payment-status').value;

            const remark = document.getElementById('remark').value.trim();


            if(!name || !weeklyRent || !dueDate) {

                alert('請填寫姓名、週租金與繳費日期！');

                return;

            }


            const data = { name, weeklyRent, moveInDate, dueDate, status, remark };


            try {

                if(id) {

                    // 更新舊資料

                    const docRef = doc(db, "tenants", id);

                    await updateDoc(docRef, data);

                } else {

                    // 新增資料

                    await addDoc(tenantsCollection, data);

                }

                clearForm();

            } catch (error) {

                alert("儲存失敗: " + error.message);

            }

        }


        // 刪除資料

        async function deleteTenant(id) {

            if(confirm('確定要刪除這筆記錄嗎？')) {

                await deleteDoc(doc(db, "tenants", id));

            }

        }


        // 帶入編輯資料

        function editTenant(id, name, rent, moveIn, due, status, remark) {

            document.getElementById('edit-id').value = id;

            document.getElementById('tenant-name').value = name;

            document.getElementById('weekly-rent').value = rent;

            document.getElementById('move-in-date').value = moveIn === 'undefined' ? '' : moveIn;

            document.getElementById('due-date').value = due;

            document.getElementById('payment-status').value = status;

            document.getElementById('remark').value = remark === 'undefined' ? '' : remark;


            document.getElementById('save-btn').innerText = "更新資料";

            document.getElementById('cancel-btn').classList.remove('hidden');

        }


        // 清空表單

        function clearForm() {

            document.getElementById('edit-id').value = '';

            document.getElementById('tenant-name').value = '';

            document.getElementById('weekly-rent').value = '';

            document.getElementById('move-in-date').value = '';

            document.getElementById('due-date').value = '';

            document.getElementById('payment-status').value = '未繳';

            document.getElementById('remark').value = '';


            document.getElementById('save-btn').innerText = "儲存資料";

            document.getElementById('cancel-btn').classList.add('hidden');

        }
